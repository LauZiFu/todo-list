
global projectList = [];
global id = 0;

function addTodo(id){  
    returnValue = {title, dueDate, priority, description};
    const newTodoItem = new TodoItem(returnValue);
    projectList[id].addItem(newTodoItem);

    localstorage.setItem(`${projectList[id].title}`, JSON.stringify(projectList[id].json()));
    renderProject(projectList[id]);
}

function createProject(dataStructure = []){
    id++;
    returnValue = {title, description};
    returnValue.dataStructure. = dataStructure;
    const newProject = new Project(returnValue);
    newProject.setId(id);

    localstorage.setItem(`${returnValue.title}`, JSON.stringify(newProject.json()))
    renderProject(projectList[id]);
}

// inside project class
function changeDS(dataStructure){
    for each item in this.itemList{
        dataStructure.insert(item);
    }
    this.itemList = dataStructure;
}
